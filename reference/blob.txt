<script src="https://cdnjs.cloudflare.com/ajax/libs/three.js/r128/three.min.js"></script>
<script src="https://cdn.jsdelivr.net/npm/three@0.128.0/examples/js/controls/OrbitControls.js"></script>

<script>
(function () {

  var params = {
    timeScale:        0.78,
    rotationSpeedX:   0.002,
    rotationSpeedY:   0.005,
    plasmaScale:      0.1404,
    plasmaBrightness: 1.31,
    voidThreshold:    0.072,
    colorDeep:   0x001a1a,
    colorMid:    0x00a8a8,
    colorBright: 0x01F2F2,
    shellColor:  0x01F2F2,
    shellOpacity: 0.41
  };

  // 1. SCENE SETUP
  var scene = new THREE.Scene();
  scene.background = new THREE.Color(0x000000);

  var camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 100);
  camera.position.z = 2.4;

  var renderer = new THREE.WebGLRenderer({ antialias: true });
  renderer.setSize(window.innerWidth, window.innerHeight);
  renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
  renderer.toneMapping = THREE.ACESFilmicToneMapping;
  renderer.toneMappingExposure = 0.9;
  document.body.appendChild(renderer.domElement);

  var controls = new THREE.OrbitControls(camera, renderer.domElement);
  controls.enableDamping = true;
  controls.enablePan = false;
  controls.minDistance = 1.5;
  controls.maxDistance = 20;

  var mainGroup = new THREE.Group();
  scene.add(mainGroup);

  // GLSL NOISE FUNCTIONS
  var noiseFunctions =
    'vec3 mod289(vec3 x){return x-floor(x*(1.0/289.0))*289.0;}\n' +
    'vec4 mod289(vec4 x){return x-floor(x*(1.0/289.0))*289.0;}\n' +
    'vec4 permute(vec4 x){return mod289(((x*34.0)+1.0)*x);}\n' +
    'vec4 taylorInvSqrt(vec4 r){return 1.79284291400159-0.85373472095314*r;}\n' +
    'float snoise(vec3 v){\n' +
    '  const vec2 C=vec2(1.0/6.0,1.0/3.0);\n' +
    '  const vec4 D=vec4(0.0,0.5,1.0,2.0);\n' +
    '  vec3 i=floor(v+dot(v,C.yyy));\n' +
    '  vec3 x0=v-i+dot(i,C.xxx);\n' +
    '  vec3 g=step(x0.yzx,x0.xyz);\n' +
    '  vec3 l=1.0-g;\n' +
    '  vec3 i1=min(g.xyz,l.zxy);\n' +
    '  vec3 i2=max(g.xyz,l.zxy);\n' +
    '  vec3 x1=x0-i1+C.xxx;\n' +
    '  vec3 x2=x0-i2+C.yyy;\n' +
    '  vec3 x3=x0-D.yyy;\n' +
    '  i=mod289(i);\n' +
    '  vec4 p=permute(permute(permute(\n' +
    '    i.z+vec4(0.0,i1.z,i2.z,1.0))\n' +
    '    +i.y+vec4(0.0,i1.y,i2.y,1.0))\n' +
    '    +i.x+vec4(0.0,i1.x,i2.x,1.0));\n' +
    '  float n_=0.142857142857;\n' +
    '  vec3 ns=n_*D.wyz-D.xzx;\n' +
    '  vec4 j=p-49.0*floor(p*ns.z*ns.z);\n' +
    '  vec4 x_=floor(j*ns.z);\n' +
    '  vec4 y_=floor(j-7.0*x_);\n' +
    '  vec4 x=x_*ns.x+ns.yyyy;\n' +
    '  vec4 y=y_*ns.x+ns.yyyy;\n' +
    '  vec4 h=1.0-abs(x)-abs(y);\n' +
    '  vec4 b0=vec4(x.xy,y.xy);\n' +
    '  vec4 b1=vec4(x.zw,y.zw);\n' +
    '  vec4 s0=floor(b0)*2.0+1.0;\n' +
    '  vec4 s1=floor(b1)*2.0+1.0;\n' +
    '  vec4 sh=-step(h,vec4(0.0));\n' +
    '  vec4 a0=b0.xzyw+s0.xzyw*sh.xxyy;\n' +
    '  vec4 a1=b1.xzyw+s1.xzyw*sh.zzww;\n' +
    '  vec3 p0=vec3(a0.xy,h.x);\n' +
    '  vec3 p1=vec3(a0.zw,h.y);\n' +
    '  vec3 p2=vec3(a1.xy,h.z);\n' +
    '  vec3 p3=vec3(a1.zw,h.w);\n' +
    '  vec4 norm=taylorInvSqrt(vec4(dot(p0,p0),dot(p1,p1),dot(p2,p2),dot(p3,p3)));\n' +
    '  p0*=norm.x;p1*=norm.y;p2*=norm.z;p3*=norm.w;\n' +
    '  vec4 m=max(0.6-vec4(dot(x0,x0),dot(x1,x1),dot(x2,x2),dot(x3,x3)),0.0);\n' +
    '  m=m*m;\n' +
    '  return 42.0*dot(m*m,vec4(dot(p0,x0),dot(p1,x1),dot(p2,x2),dot(p3,x3)));\n' +
    '}\n' +
    'float fbm(vec3 p){\n' +
    '  float total=0.0,amplitude=0.5,frequency=1.0;\n' +
    '  for(int i=0;i<3;i++){total+=snoise(p*frequency)*amplitude;amplitude*=0.5;frequency*=2.0;}\n' +
    '  return total;\n' +
    '}\n';

  // 2. LIGHTS
  var pointLight = new THREE.PointLight(0x00cccc, 2.0, 10);
  mainGroup.add(pointLight);

  // 3. OUTER SHELL — fresnel edge only, no halo
  var shellGeo = new THREE.SphereGeometry(1.0, 64, 64);

  var shellVS =
    'varying vec3 vNormal;\n' +
    'varying vec3 vViewPosition;\n' +
    'void main(){\n' +
    '  vNormal=normalize(normalMatrix*normal);\n' +
    '  vec4 mvPosition=modelViewMatrix*vec4(position,1.0);\n' +
    '  vViewPosition=-mvPosition.xyz;\n' +
    '  gl_Position=projectionMatrix*mvPosition;\n' +
    '}';

  var shellFS =
    'varying vec3 vNormal;\n' +
    'varying vec3 vViewPosition;\n' +
    'uniform vec3 uColor;\n' +
    'uniform float uOpacity;\n' +
    'void main(){\n' +
    '  float fresnel=pow(1.0-dot(normalize(vNormal),normalize(vViewPosition)),2.5);\n' +
    '  gl_FragColor=vec4(uColor,fresnel*uOpacity);\n' +
    '}';

  // Back shell removed (was the halo) — only front shell kept
  var shellFrontMat = new THREE.ShaderMaterial({
    vertexShader: shellVS,
    fragmentShader: shellFS,
    uniforms:{
      uColor:   { value: new THREE.Color(0x01F2F2) },
      uOpacity: { value: params.shellOpacity }
    },
    transparent: true,
    blending: THREE.AdditiveBlending,
    side: THREE.FrontSide,
    depthWrite: false
  });

  mainGroup.add(new THREE.Mesh(shellGeo, shellFrontMat));

  // 4. PLASMA
  var plasmaGeo = new THREE.SphereGeometry(0.998, 128, 128);

  var plasmaFS =
    'uniform float uTime,uScale,uBrightness,uThreshold;\n' +
    'uniform vec3 uColorDeep,uColorMid,uColorBright;\n' +
    'varying vec3 vPosition,vNormal,vViewPosition;\n' +
    noiseFunctions +
    'void main(){\n' +
    '  vec3 p=vPosition*uScale;\n' +
    '  vec3 q=vec3(\n' +
    '    fbm(p+vec3(0.0,uTime*0.05,0.0)),\n' +
    '    fbm(p+vec3(5.2,1.3,2.8)+uTime*0.05),\n' +
    '    fbm(p+vec3(2.2,8.4,0.5)-uTime*0.02)\n' +
    '  );\n' +
    '  float density=fbm(p+2.0*q);\n' +
    '  float t=(density+0.4)*0.8;\n' +
    '  float alpha=smoothstep(uThreshold,0.7,t);\n' +
    '  vec3 cWhite=vec3(1.0,1.0,1.0);\n' +
    '  vec3 color=mix(uColorDeep,uColorMid,smoothstep(uThreshold,0.5,t));\n' +
    '  color=mix(color,uColorBright,smoothstep(0.5,0.8,t));\n' +
    '  color=mix(color,cWhite,smoothstep(0.8,1.0,t));\n' +
    '  float facing=dot(normalize(vNormal),normalize(vViewPosition));\n' +
    '  float depthFactor=(facing+1.0)*0.5;\n' +
    '  float finalAlpha=alpha*(0.02+0.98*depthFactor);\n' +
    '  gl_FragColor=vec4(color*uBrightness,finalAlpha);\n' +
    '}';

  var plasmaMat = new THREE.ShaderMaterial({
    uniforms:{
      uTime:        { value: 0 },
      uScale:       { value: params.plasmaScale },
      uBrightness:  { value: params.plasmaBrightness },
      uThreshold:   { value: params.voidThreshold },
      uColorDeep:   { value: new THREE.Color(params.colorDeep) },
      uColorMid:    { value: new THREE.Color(params.colorMid) },
      uColorBright: { value: new THREE.Color(params.colorBright) }
    },
    vertexShader:
      'varying vec3 vPosition,vNormal,vViewPosition;\n' +
      'void main(){\n' +
      '  vPosition=position;\n' +
      '  vNormal=normalize(normalMatrix*normal);\n' +
      '  vec4 mvPosition=modelViewMatrix*vec4(position,1.0);\n' +
      '  vViewPosition=-mvPosition.xyz;\n' +
      '  gl_Position=projectionMatrix*mvPosition;\n' +
      '}',
    fragmentShader: plasmaFS,
    transparent: true,
    blending: THREE.AdditiveBlending,
    side: THREE.DoubleSide,
    depthWrite: false
  });

  var plasmaMesh = new THREE.Mesh(plasmaGeo, plasmaMat);
  mainGroup.add(plasmaMesh);

  // 5. PARTICLES
  var pCount = 600;
  var pPos   = new Float32Array(pCount * 3);
  var pSizes = new Float32Array(pCount);
  var sphereRadius = 0.95;

  for (var i = 0; i < pCount; i++) {
    var r     = sphereRadius * Math.cbrt(Math.random());
    var theta = Math.random() * Math.PI * 2;
    var phi   = Math.acos(2 * Math.random() - 1);
    pPos[i*3]   = r * Math.sin(phi) * Math.cos(theta);
    pPos[i*3+1] = r * Math.sin(phi) * Math.sin(theta);
    pPos[i*3+2] = r * Math.cos(phi);
    pSizes[i]   = Math.random();
  }

  var pGeo = new THREE.BufferGeometry();
  pGeo.setAttribute('position', new THREE.BufferAttribute(pPos, 3));
  pGeo.setAttribute('aSize',    new THREE.BufferAttribute(pSizes, 1));

  var pMat = new THREE.ShaderMaterial({
    uniforms:{
      uTime:  { value: 0 },
      uColor: { value: new THREE.Color(0xffffff) }
    },
    vertexShader:
      'uniform float uTime;\n' +
      'attribute float aSize;\n' +
      'varying float vAlpha;\n' +
      'void main(){\n' +
      '  vec3 pos=position;\n' +
      '  pos.y+=sin(uTime*0.2+pos.x)*0.02;\n' +
      '  pos.x+=cos(uTime*0.15+pos.z)*0.02;\n' +
      '  vec4 mvPosition=modelViewMatrix*vec4(pos,1.0);\n' +
      '  gl_Position=projectionMatrix*mvPosition;\n' +
      '  float baseSize=8.0*aSize+4.0;\n' +
      '  gl_PointSize=baseSize*(1.0/-mvPosition.z);\n' +
      '  vAlpha=0.8+0.2*sin(uTime+aSize*10.0);\n' +
      '}',
    fragmentShader:
      'uniform vec3 uColor;\n' +
      'varying float vAlpha;\n' +
      'void main(){\n' +
      '  vec2 uv=gl_PointCoord-vec2(0.5);\n' +
      '  float dist=length(uv);\n' +
      '  if(dist>0.5)discard;\n' +
      '  float glow=1.0-(dist*2.0);\n' +
      '  glow=pow(glow,1.8);\n' +
      '  gl_FragColor=vec4(uColor,glow*vAlpha);\n' +
      '}',
    transparent: true,
    blending: THREE.AdditiveBlending,
    depthWrite: false
  });

  var particles = new THREE.Points(pGeo, pMat);
  mainGroup.add(particles);

  // 6. ANIMATION LOOP
  var clock = new THREE.Clock();

  function animate() {
    requestAnimationFrame(animate);
    var t = clock.getElapsedTime();

    plasmaMat.uniforms.uTime.value = t * params.timeScale;
    pMat.uniforms.uTime.value      = t;

    plasmaMesh.rotation.y = t * 0.08;

    mainGroup.rotation.x += params.rotationSpeedX;
    mainGroup.rotation.y += params.rotationSpeedY;

    controls.update();
    renderer.render(scene, camera);
  }

  window.addEventListener('resize', function () {
    camera.aspect = window.innerWidth / window.innerHeight;
    camera.updateProjectionMatrix();
    renderer.setSize(window.innerWidth, window.innerHeight);
  });

  animate();

})();
</script>
